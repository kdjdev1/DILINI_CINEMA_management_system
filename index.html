-- Dilini Cinema RAD Project — Supabase Schema -- Run this entire block in Supabase SQL Editor create table if not exists movies ( id uuid default gen_random_uuid() primary key, title text not null, genre text default 'Drama', duration integer default 120, rating decimal(3,1) default 8.0, description text, cast_ text, poster_bg text default 'linear-gradient(140deg,#1a0000,#2e0a0a)', poster_color text default '#e87060', poster_url text, trailer_url text, is_active boolean default true, created_at timestamptz default now() ); create table if not exists showtimes ( id uuid default gen_random_uuid() primary key, movie_id uuid references movies(id) on delete cascade, show_date date not null, show_time time not null, screen text default 'Screen 1', total_seats integer default 180, available_seats integer default 180, price decimal(10,2) not null default 750, is_active boolean default true, created_at timestamptz default now() ); create table if not exists concessions ( id uuid default gen_random_uuid() primary key, name text not null, category text default 'Snacks', price decimal(10,2) default 0, emoji text default '🍿', stock integer default 100, is_active boolean default true, created_at timestamptz default now() ); create table if not exists customers ( id uuid default gen_random_uuid() primary key, name text not null, email text, whatsapp text unique, loyalty_points integer default 0, tier text default 'Silver', total_spent decimal(10,2) default 0, visits integer default 0, created_at timestamptz default now() ); create table if not exists bookings ( id uuid default gen_random_uuid() primary key, booking_ref text unique not null, customer_id uuid references customers(id), showtime_id uuid references showtimes(id), seats text[] default '{}', seat_count integer default 1, ticket_total decimal(10,2) default 0, concession_total decimal(10,2) default 0, grand_total decimal(10,2) default 0, loyalty_points_earned integer default 0, loyalty_points_used integer default 0, discount_amount decimal(10,2) default 0, status text default 'confirmed', created_at timestamptz default now() ); create table if not exists booking_items ( id uuid default gen_random_uuid() primary key, booking_id uuid references bookings(id) on delete cascade, item_name text, quantity integer default 1, unit_price decimal(10,2), subtotal decimal(10,2) ); -- Disable RLS for development alter table movies disable row level security; alter table showtimes disable row level security; alter table concessions disable row level security; alter table customers disable row level security;
